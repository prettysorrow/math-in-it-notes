\newenvironment{callout}[1]{%
	\begin{tcolorbox}[
			breakable,
			leftrule=4pt,
			rightrule=0pt,
			toprule=0pt,
			bottomrule=0pt,
			colback=\calloutcolorname{#1}!5,
			colframe=\calloutcolorname{#1},
			arc=0pt,
			outer arc=0pt,
			left=8pt,
			right=8pt,
			top=4pt,
			bottom=4pt,
			parbox=false,
		]
		{\normalfont\bfseries\small\color{\calloutcolorname{#1}}\MakeUppercase{#1}\par\nobreak\smallskip}
		}{%
	\end{tcolorbox}
}
